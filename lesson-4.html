Урок 4: Работа с DOM

30 минут Средний

Что такое DOM?

DOM (Document Object Model) — это программный интерфейс для HTML-документов. Он представляет страницу в виде дерева объектов, с которыми можно взаимодействовать через JavaScript.

DOM позволяет:

  • Находить элементы на странице
  • Изменять содержимое элементов
  • Добавлять и удалять элементы
  • Изменять стили элементов
  • Обрабатывать события (клики, ввод текста и т.д.)

Поиск элементов на странице


// Поиск по ID (возвращает один элемент)
const header = document.getElementById('header');

// Поиск по классу (возвращает коллекцию)
const items = document.getElementsByClassName('item');

// Поиск по тегу (возвращает коллекцию)
const paragraphs = document.getElementsByTagName('p');

// Поиск по селектору CSS (первый элемент)
const firstButton = document.querySelector('.btn');

// Поиск по селектору CSS (все элементы)
const allButtons = document.querySelectorAll('.btn');

// Примеры селекторов:
// .class - поиск по классу
// #id - поиск по id
// div p - все p внутри div
// button.btn - кнопки с классом btn

Изменение содержимого


// Изменение текста
const title = document.getElementById('title');
title.textContent = 'Новый заголовок';

// Изменение HTML содержимого (осторожно, XSS риски!)
title.innerHTML = '<strong>Жирный текст</strong>';

// Получение и изменение атрибутов
const link = document.querySelector('a');
link.getAttribute('href');                 // получить
link.setAttribute('href', 'https://example.com'); // установить
link.removeAttribute('target');            // удалить

// Работа с data-атрибутами
const card = document.querySelector('.card');
card.dataset.userId = '123';              // data-user-id="123"
console.log(card.dataset.userId);          // '123'

Это демонстрационный блок. Нажмите на кнопку ниже, чтобы изменить текст.

Изменение стилей


const box = document.querySelector('.box');

// Изменение отдельных стилей
box.style.backgroundColor = 'blue';
box.style.color = 'white';
box.style.padding = '10px';
box.style.fontSize = '18px';

// Добавление/удаление классов (рекомендуемый способ)
box.classList.add('active');           // добавить класс
box.classList.remove('active');        // удалить класс
box.classList.toggle('active');        // переключить класс
box.classList.contains('active');      // проверить наличие

// Получение вычисленных стилей
const styles = getComputedStyle(box);
console.log(styles.backgroundColor);

Нажмите на квадрат, чтобы изменить цвет

Создание и удаление элементов


// Создание элемента
const newDiv = document.createElement('div');
newDiv.textContent = 'Новый элемент';
newDiv.classList.add('new-class');

// Добавление на страницу
const container = document.getElementById('container');
container.appendChild(newDiv);              // в конец
container.insertBefore(newDiv, container.firstChild); // в начало

// Вставка HTML
container.insertAdjacentHTML('beforeend', '<p>Новый параграф</p>');

// Удаление элемента
const elementToRemove = document.getElementById('remove-me');
elementToRemove.remove();                   // современный способ
container.removeChild(elementToRemove);     // старый способ

// Клонирование элемента
const clonedElement = elementToRemove.cloneNode(true); // true - глубокое клонирование

Обработка событий

События — это действия пользователя или браузера, на которые можно реагировать.


const button = document.getElementById('myButton');

// Добавление обработчика
button.addEventListener('click', function(event) {
    console.log('Кнопка нажата!');
    console.log('Целевой элемент:', event.target);
    console.log('Тип события:', event.type);
});

// Стрелочная функция
button.addEventListener('click', (event) => {
    console.log('Клик!');
});

// Основные типы событий:
// click - клик мышью
// mouseenter / mouseleave - наведение мыши
// submit - отправка формы
// input - ввод текста
// keydown / keyup - нажатие клавиш
// scroll - прокрутка
// load - загрузка страницы

Количество кликов: 0


// Удаление обработчика
button.removeEventListener('click', handlerFunction);

// Всплытие событий (event bubbling)
document.querySelector('.parent').addEventListener('click', () => {
    console.log('Клик на родителе');
});

document.querySelector('.child').addEventListener('click', (e) => {
    e.stopPropagation(); // остановить всплытие
    console.log('Клик на ребенке');
});

// Делегирование событий
document.querySelector('#taskList').addEventListener('click', (e) => {
    if (e.target.classList.contains('delete-task')) {
        e.target.parentElement.remove();
    }
});
