Урок 3: Функции

25 минут Средний

Что такое функции?

Функции — это "строительные блоки" JavaScript. Это блоки кода, которые можно многократно использовать. Функции позволяют:

  • Избегать дублирования кода (DRY-принцип)
  • Разбивать программу на логические части
  • Упрощать отладку и поддержку кода
  • Создавать переиспользуемые компоненты

Способы создания функций

1. Function Declaration (объявление функции)

Классический способ. Функцию можно вызвать до ее объявления (hoisting).


function greet(name) {
    return "Привет, " + name + "!";
}

console.log(greet("Анна")); // "Привет, Анна!"
console.log(greet("Петр")); // "Привет, Петр!"

2. Function Expression (функциональное выражение)

Функция присваивается переменной. Вызвать можно только после объявления.


const greet = function(name) {
    return "Привет, " + name + "!";
};

console.log(greet("Мария")); // "Привет, Мария!"

3. Стрелочные функции (Arrow Functions)

Современный и лаконичный синтаксис. Не имеют собственного this.


// Полная форма
const greet = (name) => {
    return "Привет, " + name + "!";
};

// Если один параметр, скобки можно опустить
const greetShort = name => {
    return "Привет, " + name + "!";
};

// Если тело состоит из одного выражения, можно опустить return и фигурные скобки
const greetShortest = name => "Привет, " + name + "!";

// Без параметров
const sayHello = () => "Привет!";

console.log(sayHello()); // "Привет!"

Параметры и аргументы

Функции могут принимать данные через параметры и возвращать результат.


// Функция с несколькими параметрами
function sum(a, b) {
    return a + b;
}

// Функция с параметрами по умолчанию (ES6)
function multiply(a, b = 1) {
    return a * b;
}

console.log(sum(5, 3));      // 8
console.log(multiply(5));    // 5 (b = 1 по умолчанию)
console.log(multiply(5, 2)); // 10

// Функция, возвращающая несколько значений через объект
function getUserInfo(name, age, city) {
    return {
        userName: name,
        userAge: age,
        userCity: city,
        isAdult: age >= 18
    };
}

const user = getUserInfo("Иван", 25, "Москва");
console.log(user.userName); // "Иван"
console.log(user.isAdult);  // true

Область видимости (Scope)

Переменные, объявленные внутри функции, доступны только внутри этой функции (локальная область видимости).


let globalVar = "Я глобальная переменная"; // глобальная

function myFunction() {
    let localVar = "Я локальная переменная"; // локальная
    console.log(globalVar); // ✅ Доступно
    console.log(localVar);  // ✅ Доступно
}

myFunction();
console.log(globalVar);     // ✅ Доступно
console.log(localVar);      // ❌ Ошибка! localVar не определена

// Переменные с одинаковыми именами в разных функциях не конфликтуют
function first() {
    let message = "Первая функция";
    console.log(message);
}

function second() {
    let message = "Вторая функция";
    console.log(message);
}

Замыкания (Closures)

Замыкание — это функция, которая "запоминает" переменные из внешней области видимости даже после того, как внешняя функция завершила выполнение.


function createCounter() {
    let count = 0; // переменная "замыкается"
    
    return function() {
        count++;
        return count;
    };
}

const counter = createCounter();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

// Каждый вызов createCounter создает независимый счетчик
const counter2 = createCounter();
console.log(counter2()); // 1 (начинает с нуля)
console.log(counter());  // 4 (продолжает свой счет)

Функции высшего порядка

Функции, которые принимают другие функции как аргументы или возвращают функции.


// Функция, принимающая другую функцию
function calculate(a, b, operation) {
    return operation(a, b);
}

const add = (x, y) => x + y;
const multiply = (x, y) => x * y;

console.log(calculate(5, 3, add));      // 8
console.log(calculate(5, 3, multiply)); // 15

// Встроенные методы массивов - примеры функций высшего порядка
let numbers = [1, 2, 3, 4, 5];

// map - преобразует каждый элемент
let doubled = numbers.map(n => n * 2);
console.log(doubled); // [2, 4, 6, 8, 10]

// filter - фильтрует элементы
let evenNumbers = numbers.filter(n => n % 2 === 0);
console.log(evenNumbers); // [2, 4]

// reduce - сворачивает массив в одно значение
let sum = numbers.reduce((acc, n) => acc + n, 0);
console.log(sum); // 15

Практическое задание

1. Напишите функцию `isEven`, которая принимает число и возвращает true, если число четное, и false, если нечетное.

2. Напишите функцию `getMax`, которая принимает два числа и возвращает большее из них.

3. Напишите функцию `factorial`, которая вычисляет факториал числа (произведение всех чисел от 1 до n). Используйте рекурсию.

4. Напишите функцию `createMultiplier`, которая принимает число и возвращает новую функцию, умножающую свой аргумент на это число. Пример: `const double = createMultiplier(2); double(5) => 10`.
