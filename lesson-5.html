Урок 5: Асинхронность

35 минут Продвинутый

Что такое асинхронность?

JavaScript — однопоточный язык. Это значит, что он может выполнять только одну операцию за раз. Асинхронность позволяет выполнять длительные операции (запросы к серверу, таймеры, чтение файлов) без блокировки основного потока выполнения.

Представьте ресторан: официант (главный поток) принимает заказы, но не ждет, пока приготовят еду — он передает заказ на кухню и продолжает обслуживать других посетителей. Когда еда готова, официант возвращается к столику. Так же работает асинхронный JavaScript.

setTimeout и setInterval

Простейшие способы создания асинхронного кода.


// setTimeout - выполнить код через указанное время (в миллисекундах)
setTimeout(() => {
    console.log('Это сообщение появится через 2 секунды');
}, 2000);

console.log('Это сообщение появится сразу');

// Результат:
// "Это сообщение появится сразу"
// (пауза 2 секунды)
// "Это сообщение появится через 2 секунды"

// setInterval - выполнять код периодически
let counter = 0;
const interval = setInterval(() => {
    counter++;
    console.log(`Прошло ${counter} секунд`);
    
    if (counter === 5) {
        clearInterval(interval); // остановить интервал
        console.log('Интервал остановлен');
    }
}, 1000);

Таймер в действии

Секундомер: 0 сек

Callback функции (колбэки)

Колбэк — это функция, которая передается в другую функцию и вызывается после завершения асинхронной операции.


// Пример функции с колбэком
function loadData(callback) {
    console.log('Начинаем загрузку данных...');
    
    setTimeout(() => {
        const data = { 
            name: "Иван", 
            age: 25,
            city: "Москва" 
        };
        callback(data);
    }, 2000);
}

// Использование
loadData((result) => {
    console.log("Данные загружены:", result);
});

// Проблема колбэков: "callback hell" (ад колбэков)
function getUser(callback) {
    setTimeout(() => {
        callback({ id: 1, name: "Иван" });
    }, 1000);
}

function getOrders(userId, callback) {
    setTimeout(() => {
        callback(["Заказ 1", "Заказ 2"]);
    }, 1000);
}

function getOrderDetails(order, callback) {
    setTimeout(() => {
        callback({ price: 1000, items: 2 });
    }, 1000);
}

// Ад колбэков - трудно читать и поддерживать
getUser((user) => {
    getOrders(user.id, (orders) => {
        getOrderDetails(orders[0], (details) => {
            console.log(details);
        });
    });
});

Promise

Promise — это объект, представляющий результат асинхронной операции, которая может быть выполнена сейчас, в будущем или никогда. Promise решает проблему "ада колбэков".

Состояния Promise:

  • pending — ожидание (начальное состояние)
  • fulfilled — выполнено успешно (resolve)
  • rejected — отклонено (ошибка, reject)

// Создание Promise
const promise = new Promise((resolve, reject) => {
    setTimeout(() => {
        const success = true;
        
        if (success) {
            resolve("Операция выполнена успешно");
        } else {
            reject("Произошла ошибка");
        }
    }, 1000);
});

// Использование Promise
promise
    .then(result => {
        console.log("Успех:", result);
    })
    .catch(error => {
        console.error("Ошибка:", error);
    })
    .finally(() => {
        console.log("Операция завершена (в любом случае)");
    });

// Цепочка Promise
function getUser() {
    return new Promise(resolve => {
        setTimeout(() => resolve({ id: 1, name: "Иван" }), 1000);
    });
}

function getOrders(userId) {
    return new Promise(resolve => {
        setTimeout(() => resolve(["Заказ 1", "Заказ 2"]), 1000);
    });
}

function getOrderDetails(order) {
    return new Promise(resolve => {
        setTimeout(() => resolve({ price: 1000, items: 2 }), 1000);
    });
}

// Чистая цепочка Promise (без ада)
getUser()
    .then(user => {
        console.log("Пользователь:", user);
        return getOrders(user.id);
    })
    .then(orders => {
        console.log("Заказы:", orders);
        return getOrderDetails(orders[0]);
    })
    .then(details => {
        console.log("Детали заказа:", details);
    })
    .catch(error => {
        console.error("Ошибка:", error);
    });

Promise.all и Promise.race


// Promise.all - ждет выполнения всех Promise
const promise1 = Promise.resolve(3);
const promise2 = 42;
const promise3 = new Promise((resolve) => {
    setTimeout(resolve, 100, 'foo');
});

Promise.all([promise1, promise2, promise3]).then((values) => {
    console.log(values); // [3, 42, "foo"]
});

// Promise.race - возвращает первый выполнившийся Promise
const p1 = new Promise(resolve => setTimeout(resolve, 500, 'быстрый'));
const p2 = new Promise(resolve => setTimeout(resolve, 1000, 'медленный'));

Promise.race([p1, p2]).then(value => {
    console.log(value); // "быстрый"
});

Демонстрация Promise

async/await

Современный и самый удобный способ работы с асинхронным кодом. async/await делает асинхронный код похожим на синхронный.


// Объявление асинхронной функции
async function fetchData() {
    try {
        console.log("Начинаем загрузку...");
        
        const user = await getUser();
        console.log("Загружен пользователь:", user);
        
        const orders = await getOrders(user.id);
        console.log("Загружены заказы:", orders);
        
        const details = await getOrderDetails(orders[0]);
        console.log("Детали заказа:", details);
        
        return { user, orders, details };
    } catch (error) {
        console.error("Ошибка:", error);
    }
}

// Вызов асинхронной функции
fetchData().then(result => {
    console.log("Все данные:", result);
});

// Обработка ошибок с try...catch
async function loadData() {
    try {
        const response = await fetch('https://api.example.com/data');
        const data = await response.json();
        return data;
    } catch (error) {
        console.error("Ошибка загрузки:", error);
        return null;
    }
}

Работа с API (Fetch)

Fetch API — современный способ делать HTTP-запросы к серверам и API.


// GET запрос
async function getUsers() {
    try {
        const response = await fetch('https://jsonplaceholder.typicode.com/users');
        
        if (!response.ok) {
            throw new Error(`HTTP ошибка: ${response.status}`);
        }
        
        const users = await response.json();
        console.log(users);
        return users;
    } catch (error) {
        console.error("Ошибка загрузки:", error);
    }
}

// POST запрос
async function createUser(userData) {
    try {
        const response = await fetch('https://jsonplaceholder.typicode.com/users', {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify(userData)
        });
        
        const result = await response.json();
        console.log("Создан пользователь:", result);
        return result;
    } catch (error) {
        console.error("Ошибка создания:", error);
    }
}

// Пример использования
async function demo() {
    // Получаем пользователей
    const users = await getUsers();
    console.log(`Загружено ${users.length} пользователей`);
    
    // Создаем нового пользователя
    const newUser = await createUser({
        name: "Иван Петров",
        email: "ivan@example.com"
    });
}
